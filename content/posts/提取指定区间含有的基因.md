```bash
# 根据bed文件和gff提取bed区间里头含有什么基因
bedtools intersect -a a.bed -b /home2/pubdata/ref_data/Lchinesis_genome.Chr.gff3 -wa -wb > a.bed_intersect_results.txt
bedtools intersect -a j.bed -b /home2/pubdata/ref_data/Lchinesis_genome.Chr.gff3 -wa -wb > j.bed_intersect_results.txt
bedtools intersect -a w.bed -b /home2/pubdata/ref_data/Lchinesis_genome.Chr.gff3 -wa -wb > w.bed_intersect_results.txt
```

[bedtools intersect用法 (intersectBed) - 简书 (jianshu.com)](https://www.jianshu.com/p/a2cff67c38a9)

#### 提取基因区间的含有的基因的ID

```bash
less -S j.bed_intersect_results.txt | grep "mRNA" | cut -f12 | sort -u > j.bed.gene.txt
less -S a.bed_intersect_results.txt | grep "mRNA" | cut -f12 | sort -u > a.bed.gene.txt
less -S w.bed_intersect_results.txt | grep "mRNA" | cut -f12 | sort -u > w.bed.gene.txt

less -S w.bed.gene.txt | sed 's/\;.*//' | cut -d '.' -f 1 | sort -u > temp && mv temp w.bed.gene.txt
less -S a.bed.gene.txt | sed 's/\;.*//' | cut -d '.' -f 1 | sort -u > temp && mv temp a.bed.gene.txt
less -S j.bed.gene.txt | sed 's/\;.*//' | cut -d '.' -f 1 | sort -u > temp && mv temp j.bed.gene.txt
```
